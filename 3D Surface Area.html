     #include<stdio.h>
using namespace std;

int T[200][200];

int main() {
    int H, W;
    while(cin >> H >> W) {
        for(int i=0; i< H; i++)
            for(int j=0; j< W; j++)
                cin >> T[i][j];

        int total = 2*H*W;
        for(int i=0; i< H; i++) {
            total += T[i][0];
            for(int j=1; j< W; j++)
                total += abs(T[i][j] - T[i][j-1]);
            total += T[i][W-1];
        }
        for(int i=0; i< W; i++) {
            total += T[0][i];
            for(int j=1; j< H; j++)
                total += abs(T[j][i] - T[j-1][i]);
            total += T[H-1][i];
        }

        cout << total << endl;
    }
}

      int W = in.nextInt();
      int[][] A = new int[H][W];

          for(int A_j = 0; A_j < W; A_j++){
              A[A_i][A_j] = in.nextInt();
          }
      }
      int result = 2 * H * W;
      for (int i = 0; i < H; i++) {
    for (int j = 0; j < W; j++) {

      result += (A[i][j]) * 4;

      if(i != 0) {
        result -= 2 * Math.min(A[i - 1][j], A[i][j]);
      }
      if(j != 0) {
        result -= 2 * Math.min(A[i][j], A[i][j - 1]);
      }

  }

  def surfaceArea(A):
      H, W = len(A), len(A[0])
      area = 2*H*W
      #print("H = {} W = {} area = {}".format(len(A), len(A[0]), area))

      for ind in range(H):
          for jnd in range(W):
              #print("i = {} j = {} A = {}".format(ind, jnd, A[ind][jnd]))

                  #print("adding full {}".format(A[ind][jnd]))
                  area += A[ind][jnd]

                  area += max(0, A[ind][jnd] - A[ind][jnd-1])

                  area += A[ind][jnd]

                  area += max(0, A[ind][jnd] - A[ind][jnd+1])
